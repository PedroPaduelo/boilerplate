📽 3. Auth Store — Zustand com Persist

Interface do Store

// src/features/auth/store.ts
interface AuthState {
  // Estado
  user:            User | null
  token:           string | null
  isAuthenticated: boolean
  isLoading:       boolean
  isHydrated:      boolean  // Zustand persist rehidratou?

  // Ações
  setAuth:    (user: User, token: string) => void
  setUser:    (user: User) => void
  logout:     () => void
  setLoading: (loading: boolean) => void
  setHydrated:() => void
}

Comportamento do Persist

persist({...}, {
  name: 'auth',           // chave localStorage
  partialize: (s) => ({
    token: s.token     // persiste só token
  }),
  onRehydrateStorage: () => (s) => {
    s?.setHydrated()    // sinaliza rehidratado
  }
})

Importante: isHydrated

Quando a app carrega, o Zustand lê o localStorage de forma assíncrona. Enquanto isHydrated = false, o ProtectedRoute exibe um Skeleton em vez de redirecionar — evita flash de redirect incorreto.

🔗 4. API Client — Axios + Interceptors

Código do api-client.ts

// src/shared/lib/api-client.ts
import axios from 'axios'

const API_URL = import.meta.env.VITE_API_URL
                || 'http://localhost:3001'

export const apiClient = axios.create({
  baseURL: API_URL,
  headers: { 'Content-Type': 'application/json' }
})

// REQUEST: injeta Bearer token
apiClient.interceptors.request.use((config) => {
  const token = localStorage.getItem('token')
  if (token) {
    config.headers.Authorization = `Bearer ${token}`
  }
  return config
})

// RESPONSE: trata 401 globalmente
apiClient.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('token')
      window.location.href = '/login'
    }
    return Promise.reject(error)
  }
)

🔒 6. ProtectedRoute — Lógica de Guarda 

// src/features/auth/components/protected-route.tsx
export function ProtectedRoute({ children }) {
  const location = useLocation()
  const { token, isHydrated } = useAuthStore()

  // 1. Zustand ainda hidratando localStorage
  if (!isHydrated) {
    return <Skeleton />  // loading state
  }

  // 2. Sem token → redireciona
  if (!token) {
    return (
      <Navigate
        to="/login"
        state={{ from: location }}
        replace
      />
    )
  }

  // 3. Token ok → renderiza children
  return <>{children}</>
}

🌙 7. Dark Mode

Implementação

// src/features/dashboard/index.tsx
const [darkMode, setDarkMode] = useState(
  () => document.documentElement.classList
           .contains('dark')
)

// Toggle via Switch (shadcn/ui)
<Switch
  checked={darkMode}
  onCheckedChange={(checked) => {
    setDarkMode(checked)
    if (checked) {
      document.documentElement.classList.add('dark')
      localStorage.setItem('theme', 'dark')
    } else {
      document.documentElement.classList.remove('dark')
      localStorage.setItem('theme', 'light')
    }
  }}
/>
Config: Tailwind configurado com darkMode: 'class' — dark mode ativado pela presença da classe dark no elemento <html>.
Persist: Preferência salva em localStorage.theme como 'dark' ou 'light'.
Default: HTML inicial com classe dark aplicada por padrão neste documento e configurável no index.html.

Padrão de useQuery

// useCurrentUser — habilitado só com token
const { data, isLoading } = useQuery({
  queryKey: ['auth', 'me'],
  queryFn:  () => authApi.getMe(),
  enabled:  !!token,   // condicional
  retry:    false,
  staleTime: 1000 * 60 * 5,  // 5 min
})

// Dashboard — queryKey com range
const { data } = useQuery({
  queryKey: ['dashboard', 'timeSeries', dateRange],
  queryFn:  () => dashboardApi.getTimeSeriesData(dateRange),

Padrão de useMutation

// useLogin — mutation com side effects
export function useLogin() {
  const navigate  = useNavigate()
  const { setAuth } = useAuthStore()

  return useMutation({
    mutationFn: (input) => authApi.login(input),
    onSuccess: (data) => {
      setAuth(data.user, data.token) // Zustand
      toast.success('Login realizado!')
      navigate('/dashboard')
    },
    onError: () => {
      toast.error('Credenciais inválidas')
    }
  })
}
