export async function authenticate(socket, next) {
  const token = socket.handshake.auth.token;
  const decoded = jwt.verify(token, env.JWT_SECRET);
  (socket as any).user_id = decoded.sub; // Attach user ID
  next();
}

💻 Código Exemplo Completo

Backend - Servidor Socket.IO

// server.ts - setupSocketIO()
import { Server } from 'socket.io';
import { authenticate } from './middlewares/auth-socket';
import { socketManager } from './socket/manager/socket-manager';

export function setupSocketIO(app: FastifyInstance) {
  const io = new Server(app.server, {
    cors: {
      origin: ['http://localhost:5173'],
      credentials: true,
    },
  });

  socketManager.setIO(io);
  io.use(authenticate);

  io.on('connection', async (socket) => {
    try {
      socketManager.addSocket(socket);

      registerJoinRoom(socket);
      registerLeaveRoom(socket);
      registerDisconnect(socket);

      socket.emit('connected', {
        socketId: socket.id,
        timestamp: new Date().toISOString(),
      });
    } catch (error) {
      console.error('Error in socket connection:', error);
      socket.disconnect(true);
    }
  });

  return io;
}

Frontend - Cliente Socket.IO (Exemplo)

// src/lib/socket.ts - Frontend (NÃO IMPLEMENTADO NO PROJETO)
import { io, Socket } from 'socket.io-client';

class SocketClient {
  private socket: Socket | null = null;

  connect(token: string) {
    this.socket = io(import.meta.env.VITE_WS_URL, {
      auth: { token },
      transports: ['websocket', 'polling']
    });

    this.socket.on('connect', () => {
      console.log('✅ Connected:', this.socket?.id);
    });

    this.socket.on('connected', (data) => {
      console.log('Socket connected:', data);
    });

    this.socket.on('error', (error) => {
      console.error('Socket error:', error);
    });
  }

  joinRoom(roomId: string) {
    this.socket?.emit('join-room', roomId);
  }

  leaveRoom(roomId: string) {
    this.socket?.emit('leave-room', roomId);
  }

  disconnect() {
    this.socket?.disconnect();
    this.socket = null;
  }
}

export const socketClient = new SocketClient();

⚠️ Riscos e Vulnerabilidades

🔥 Memory Leaks

  • • Rooms órfãs: Se um cliente desconecta sem chamar leave-room, a room pode manter referências. Node.js 100% OK - socket.leave é automático no disconnect nativo
  • • Maps crescendo: roomSockets.removeSocket() limpa Maps, mas falta validação de null em get()
  • • Referências circulares: ConnectedSocket guarda Socket (que referencia SocketManager via closures)

🔓 Auth Bypass

  • • JWT verification: Usa jwt.verify() - ok se JWT_SECRET forte
  • • Race condition: Não valida se userId existe no BD após JWT decode
  • • Socket property: Usa (socket as any).user_id - possível overwrite? Sim, se outro middleware tocar
  • • Token replay: Não há blacklist de tokens revogados

🏁 Race Conditions

  • • getStats concorrente: Maps não são thread-safe em JS (single-thread ok), mas callbacks assíncronos podem ler estado inconsistente
  • • join/leave paralelo: Usuário envia join e leave quase simultâneo - o Set do socket pode ter race se não for atômico (JS single thread previne mas careful)
  • • disconnect during emit: sendToUser itera sobre userSocketIds mas o socket pode ter sido removido do connectedSockets - connectedSocket check protege

🚀 Scaling Limitations

  • • Single instance memory: socketManager é singleton na memória - não funciona em cluster/FPM
  • • No Redis adapter: Rooms são locais apenas. Multi-instância? Cada instância tem seu próprio isolation
  • • Sticky sessions: Load balancer deve usar sticky sessions (same server) ou fail

🚀 Escalabilidade com Múltiplas Instâncias

Problema Atual

O sistema atual usa socket-manager.ts singleton em memória. Se rodar em múltiplos containers/instâncias, cada uma terá seu próprio isolamento:

  • Instância A: socket.user1 em room "chat-1"
  • Instância B: socket.user2 em room "chat-1"
  • Resultado: user1 não vê mensagens de user2 (rooms são locais)

Solução: Redis Adapter

// socket.ts - Com Redis Adapter
import { createAdapter } from '@socket.io/redis-adapter';
import { Redis } from 'ioredis';

const pubClient = new Redis({ host: 'redis', port: 6379 });
const subClient = pubClient.duplicate();

io.adapter(createAdapter(pubClient, subClient));

// Agora rooms são compartilhadas entre instâncias!

📡 Pub/Sub

Todos os eventos de room são publicados no Redis e subscritos por todas as instâncias

🔗 Room State

Socket.IO gerencia membership via Redis. join-room em qualquer instância sincroniza

⚖️ Load Balancer

Agora pode usar round-robin sem sticky sessions (embora sticky ainda melhora performance)

✅ Recomendações de Implementação

1

Implementar Redis Adapter

Adicionar @socket.io/redis-adapter e configurar pub/sub Redis. Redis já existe no projeto (fastify-redis). Reutilizar conexão.

2

Implementar Frontend Socket

Criar módulo Socket.io client no frontend (ex: src/lib/socket.ts) com hook React para gerenciar conexão e estado. Conectar após login.

3

Rate Limit por Socket

Adicionar rate limiting para eventos (ex: max 10 join-room/segundo por userId) para evitar DoS.

4

Validação de Room no BD

Verificar se userId tem permissão para entrar na room (ex: roomId pertence a projeto que user tem acesso).

5

Health Check de Socket

Criar endpoint /socket/health que retorna estatísticas: total connections, rooms ativas, users online.

🔍 Insights do Código

📊 Patterns Identificados

  • Singleton Pattern - SocketManager.getInstance() para acesso global
  • Three-way Mapping - user→sockets, room→sockets, socket→metadata
  • Sentinel Value - io: Server|null (null until setIO)
  • Lazy Cleanup - Auto-remove de room quando Set vazio
  • Boolean API - Métodos retornam boolean para indicar sucesso

🐛 Bugs Potenciais

  • 1. socket as any - TypeScript bypass. Usar interface extends Socket
  • 2. Null check faltando em getStats() se Maps vazios? Na verdade OK
  • 3. Race no setIO - Se connection acontece antes de setIO, getIO() retorna null em sendToRoom/broadcast
  • 4. Timestamp UTC - connectedAt como Date, mas enviar ISO string? Pode perder timezone info
